$val = $ctr.CounterSamples.CookedValue
invoke-sqlcmd -serverinstance PS2 -Database master -Query "insert into [RDX_PS_stats] select 'PageSplits_sec' as Metric, $val as [value], GETDATE() as EventTime, DATEPART(dd, GETDATE()) as EventTime_Day, DATEPART(hh, GETDATE()) as EventTime_Hour"

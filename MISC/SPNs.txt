Grant SQL Server service account [domain\svc_account] permissions in AD.
Read servicePrincipalName
Write servicePrincipalName

/* Check for Kerberos or NTLM */
SELECT auth_scheme
FROM sys.dm_exec_connections
WHERE session_id = @@SPID;

setspn -l server

#Add SPNs#
setspn -s MSSQLSvc/server.domain.com:1433 domain\svc_account
setspn -s MSSQLSvc/server.domain.com domain\svc_account

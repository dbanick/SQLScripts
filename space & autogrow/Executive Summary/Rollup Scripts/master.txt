--io and stalls
create table #io_stalls
(drive varchar(5), 
io_stall_read_ms_per_hour bigint,
total_num_reads bigint,
total_io_stall_write_ms bigint,
total_num_writes bigint,
total_io bigint,
EventTime_day int,
EventTime_hour int
);
with CTE as (select substring([physical_name],0,3) as drive, SUM(io_stall_read_ms) as total_io_stall_read_ms, 
SUM(num_of_reads) as total_num_reads, SUM(io_stall_write_ms) as total_io_stall_write_ms, SUM(num_of_writes) as total_num_writes, 
SUM(total_io) as total_io, EventTime_Day, EventTime_Hour
from RDX_IO_Stats_2
group by substring([physical_name],0,3), EventTime_Day, EventTime_Hour
)
insert into #io_stalls
select a.drive, 
a.total_io_stall_read_ms - b.total_io_stall_read_ms as io_stall_read_ms_per_hour, 
a.total_num_reads - b.total_num_reads as total_num_reads,
a.total_io_stall_write_ms - b.total_io_stall_write_ms as total_io_stall_write_ms,
a.total_num_writes - b.total_num_writes as total_num_writes,
a.total_io - b.total_io as total_io,
a.EventTime_Day, a.EventTime_Hour
from CTE as a
inner join CTE as b
on a.drive = b.drive and a.EventTime_Day = b.EventTime_Day and a.EventTime_hour -1 = b.EventTime_Hour

--select * from #io_stalls

insert into master..[RDX_Stats_Summary]
select 'IO: Total IO Read Stalls (ms) - ' + drive,
SUM(io_stall_read_ms_per_hour) as MetricValue_Total,
AVG(io_stall_read_ms_per_hour) as MetricValue_AVG, 
MAX(io_stall_read_ms_per_hour) as MetricValue_Peak,
[EventTime_Day], [EventTime_Hour]  , GETDATE()
FROM #io_stalls
GROUP BY drive, [EventTime_Day], [EventTime_Hour]

insert into master..[RDX_Stats_Summary]
select 'IO: Total Number of Reads - ' + drive,
SUM(total_num_reads) as MetricValue_Total,
AVG(total_num_reads) as MetricValue_AVG, 
MAX(total_num_reads) as MetricValue_Peak,
[EventTime_Day], [EventTime_Hour]  , GETDATE()
FROM #io_stalls
GROUP BY drive, [EventTime_Day], [EventTime_Hour]


insert into master..[RDX_Stats_Summary]
select 'IO: Total IO Write Stalls (ms) - ' + drive,
SUM(total_io_stall_write_ms) as MetricValue_Total,
AVG(total_io_stall_write_ms) as MetricValue_AVG, 
MAX(total_io_stall_write_ms) as MetricValue_Peak,
[EventTime_Day], [EventTime_Hour]  , GETDATE()
FROM #io_stalls
GROUP BY drive, [EventTime_Day], [EventTime_Hour]

insert into master..[RDX_Stats_Summary]
select 'IO: Total Number of Writes - ' + drive,

total_MB_written dec(10,2),

a.total_MB_RW - b.total_MB_RW as total_MB_RW,
a.EventTime_Day, a.EventTime_Hour
from CTE as a
inner join CTE as b
on a.drive = b.drive and a.EventTime_Day = b.EventTime_Day and a.EventTime_hour -1 = b.EventTime_Hour

--lock waits
create table #lock_waits
(database_name varchar(150), 
[total_row_lock_waits] bigint,
[total_row_lock_waits_in_ms] bigint,
[total_page_lock_waits] bigint,
[total_page_lock_wait_in_ms] bigint,
[total_page_io_latch_lock_waits] bigint,
[total_page_io_latch_lock_wait_in_ms] bigint,
[total_lock_wait_in_ms] bigint,
EventTime_day int,
EventTime_hour int
);
with CTE as (
select database_name, SUM([total_row_lock_waits]) as [total_row_lock_waits], 
SUM([total_row_lock_wait_in_ms]) as [total_row_lock_waits_in_ms],
SUM([total_page_lock_waits]) as [total_page_lock_waits],
SUM([total_page_lock_wait_in_ms]) as [total_page_lock_wait_in_ms],
SUM([total_page_io_latch_lock_waits]) as [total_page_io_latch_lock_waits],
SUM([total_page_io_latch_lock_wait_in_ms]) as [total_page_io_latch_lock_wait_in_ms],
SUM([total_lock_wait_in_ms]) as [total_lock_wait_in_ms],
EventTime_Day, EventTime_Hour
from [dbo].[RDX_lock_wait_stats]
group by database_name, EventTime_Day, EventTime_Hour
)
insert into #lock_waits
select a.database_name, 
a.[total_row_lock_waits] - b.[total_row_lock_waits] as [total_row_lock_waits], 
a.[total_row_lock_waits_in_ms] - b.[total_row_lock_waits_in_ms] as [total_row_lock_waits_in_ms],
a.[total_page_lock_waits] - b.[total_page_lock_waits] as [total_page_lock_waits],
a.[total_page_lock_wait_in_ms] - b.[total_page_lock_wait_in_ms] as [total_page_lock_wait_in_ms],
a.[total_page_io_latch_lock_waits] - b.[total_page_io_latch_lock_waits] as [total_page_io_latch_lock_waits],
a.[total_page_io_latch_lock_wait_in_ms] - b.[total_page_io_latch_lock_wait_in_ms] as [total_page_io_latch_lock_wait_in_ms],
a.[total_lock_wait_in_ms] - b.[total_lock_wait_in_ms] as [total_lock_wait_in_ms],
a.EventTime_Day, a.EventTime_Hour 
from CTE as a
inner join CTE as b
on a.database_name = b.database_name and a.EventTime_Day = b.EventTime_Day and a.EventTime_hour -1 = b.EventTime_Hour


insert into master..[RDX_Stats_Summary]
select 'WAIT: Row Lock Waits (#) - ' + database_name,
SUM([total_row_lock_waits]) as MetricValue_Total,
AVG([total_row_lock_waits]) as MetricValue_AVG, 
MAX([total_row_lock_waits]) as MetricValue_Peak,
[EventTime_Day], [EventTime_Hour]  , GETDATE()
FROM #lock_waits
GROUP BY database_name, [EventTime_Day], [EventTime_Hour]

insert into master..[RDX_Stats_Summary]
